\section{Question No 2.43c}
\item In the following case, determine whether the given planes are parallel or perpendicular, and in case they are neither, find the angle between them.
\begin{enumerate}
\item 
$
\myvec{2 & -2 & 4}\vec{x}=-5
$
 and 
$
\myvec{3 & -3 & 6}\vec{x}=1
$ 
\end{enumerate}
\section{Solution:}
\begin{enumerate}
\item Given that,
\begin{align}
\vec{a}=\myvec{2 &-2 & 4}, 
\vec{b}=\myvec{3 & -3 &6} \label{eq1}
\end{align}
\item The given plans are  parallel if, 
\begin{align}
\frac{a_1}{b_1}=\frac{a_2}{b_2}=\frac{a_3}{b_3} \label{eq2}
\end{align}
\\
\item The given planes are perpendicular if,
\begin{align}
a. b=0
\end{align}
\\
\item Find out the given planes are parallel or perpendicular:
\\
Now using, \eqref{eq1} and \eqref{eq2} 
\begin{align}
\frac{a_1}{b_1}=\frac{a_2}{b_2}=\frac{a_3}{b_3}
\implies \frac{2}{3}=\frac{-2}{-3}=\frac{4}{6}
\\
\implies \frac{2}{3}=\frac{2}{3}=\frac{2}{3}
\end{align}
\\  
 The given planes are parallel.  So  the angle between 0\degree.
\end{enumerate}
